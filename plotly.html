Loading and filtering data

data("ny_noaa")
ny_noaa = ny_noaa %>%
  drop_na() %>%
  separate(date, into = c("year", "month", "day")) %>%
  filter(!(prcp == 0)) %>%
  filter(!(snow == 0)) %>%
  filter(!(snwd == 0)) %>%
  filter(year == 2010)

Creating bar chart

plot_ly(ny_noaa, x = ~month, y = ~snow, type = "bar")

Creating scatter plot
